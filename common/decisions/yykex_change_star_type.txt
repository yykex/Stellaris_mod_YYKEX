	sound = event_default
	icon = default
	enactment_time = 0

	resources = {
		category = decisions
	}

	potential = {
		exists = from
		from = {
			is_ai = no
			has_country_flag = yykex
        }
		is_star = yes
        is_planet_class = pc_black_hole
	}
    effect = {
		set_planet_flag = shroud_storm_changed
		change_pc = pc_b_star
		add_modifier = { modifier = shroud_switched days = -1 }
		solar_system = { set_star_class = sc_b }
    }

}
