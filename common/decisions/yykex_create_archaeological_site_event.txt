yyk_ancrel_4000 = {  #石中低语考古遗迹
	sound = event_default
	icon = default
	enactment_time = 0

	resources = {
		category = decisions
	}

	potential = {
		exists = owner 
		owner = {
			is_ai = no
			has_country_flag = yykex
        }
		is_capital = no
		NOT = {
			any_country = {
				has_country_flag = discovered_lithoids_arc_site
			}
		}
		
	}
	allow ={
		num_districts = { type = district_mining value > 1 }
		NOT = { exists = archaeological_site }
		NOT = { has_planet_flag = suppress_archaeological_sites }
		owner = { has_technology = tech_mining_3 }
	}
    effect = {
		owner = { set_country_flag = discovered_lithoids_arc_site }
		planet_event = { id = ancrel.4000 days = 0 random = 0 }
    }
	
}
