yyk_add_deposit_close = {
	sound = event_default
	icon = default
	enactment_time = 0

	resources = {
		category = decisions
	}

	potential = {
		exists = owner
		owner = {
			is_ai = no
			has_country_flag = yykex
			NOT = { has_country_flag = yykex_add_deposit_close }
        }
		is_capital = yes
	}
    effect = {
		owner = { set_country_flag = yykex_add_deposit_close }
    }

}

yyk_add_d_artifacts_3 = {
	sound = event_default
	icon = default
	enactment_time = 0

	resources = {
		category = decisions
	}

	potential = {
		exists = from
		from = {
			is_ai = no
			has_country_flag = yykex
        }
		OR = {
			AND = {
				NOT = { has_anomaly = yes }
				has_planet_flag = precursor_world
			}
			exists = archaeological_site
		}
		NOT = { has_deposit = d_artifacts_mining_3 }
		NOT = { has_deposit = d_artifacts_planet_3 }
	}

	effect = {
		if = {
			limit = { habitable_planet = yes }
			remove_deposit = d_artifacts_planet_1
			remove_deposit = d_artifacts_planet_2
			add_deposit = d_artifacts_planet_3
		}
		else = {
			every_deposit = {
				remove_deposit = yes
			}
			add_deposit = d_artifacts_mining_3
		}
	}
}
