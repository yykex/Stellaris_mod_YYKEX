yyk_crisis_10 = {
	sound = event_default
	icon = default
	enactment_time = 0

	resources = {
		category = decisions
	}

	potential = {
		exists = from
		from = {
			is_ai = no
			has_country_flag = yykex
        }
        is_star = yes
        solar_system = { is_capital_system = yes }
        NOT = { has_global_flag = prethoryn_invasion_happened }
	}

    allow={
        OR = {
            AND = { # Super early start prereqs
				end_game_years_passed >= 0
				OR = {
					has_global_flag = endgame_crisis_early_start #From Ultima Vigilis
					AND = { #Both crisis summoning resolutions active
						is_active_resolution = 	resolution_galacticstudies_extradimensional_experimentation
						is_active_resolution = resolution_galactic_threats_committee
					}
				}
			}
            AND = { # Early start
				end_game_years_passed >= 25
				OR = {
					default_endgame_early_start_triggers = yes
					any_country = {
						is_country_type = default
						OR = {
							has_technology = tech_jump_drive_1
							has_technology = tech_psi_jump_drive_1
						}
					}
					is_active_resolution = resolution_galacticstudies_extradimensional_experimentation
					is_active_resolution = resolution_galactic_threats_committee
				}
			}
			end_game_years_passed >= 50 #Regular Start
			has_global_flag = yykex_test
        }
    }

    effect = {
		every_system = {
			limit = { has_star_flag = crisis_spawn_exclude }
			set_star_flag = yykex_crisis_spawn_exclude
		}
		every_system = {
			limit = {
				NOR = {
					has_star_flag = yykex_crisis_spawn_exclude
					has_star_flag = yykex_prethoryn
				}
			}
			set_star_flag = crisis_spawn_exclude
		}
		random_country = {
            limit = { is_country_type = global_event }
            country_event = { id = crisis.10 days = 0 random = 0 }
			set_global_flag = galactic_crisis_happened
			set_timed_global_flag = {
				flag = galactic_crisis_recently_fired
				years = 12
			}
        }
		
    }

}
