yyk_spawn_storm_close = {
	sound = event_default
	icon = default
	enactment_time = 0

	resources = {
		category = decisions
	}

	potential = {
		exists = owner
		owner = {
			is_ai = no
			has_country_flag = yykex
        }
        is_capital = yes
		owner = { NOT = { has_country_flag = yykex_spawn_storm_stop } }
	}
    effect = {
		owner = {
			set_country_flag = yykex_spawn_storm_stop
		}
    }

}

yyk_spawn_nexus_storm_on_friend = {
	sound = event_default
	icon = default
	enactment_time = 0

	resources = {
		category = decisions
	}

	potential = {
		exists = from
		from = {
			is_ai = no
			has_country_flag = yykex
        }
        is_star = yes
        solar_system = { is_capital_system = yes }
	}
    effect = {
		random_system = {
			limit = {
				owner = { has_country_flag = yykex_friend }
				is_capital_system = yes
			}
			if ={
				limit = { is_inside_storm = yes }
				every_cosmic_storm = {
					limit = {any_system_within_storm = { is_same_value = prevprev } } 
					destroy_cosmic_storm = yes 
				}
			}
			spawn_random_storm = {
				type = nexus_storm
				immediate = yes 
				cosmic_storm_start_position = this
			}
		}
    }

}
