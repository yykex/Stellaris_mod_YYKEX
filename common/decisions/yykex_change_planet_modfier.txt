yyk_add_planet_modifier_close = {
	sound = event_default
	icon = default
	enactment_time = 0

	resources = {
		category = decisions
	}

	potential = {
		exists = owner
		owner = {
			is_ai = no
			has_country_flag = yykex
			NOT = { has_country_flag = yykex_add_planet_modifier_close }
        }
		is_capital = yes
	}
    effect = {
		owner = { set_country_flag = yykex_add_planet_modifier_close }
    }

}
