yyk_create_fallen_empire_1_fleet = {   #知管
    sound = event_default
	icon = default
	enactment_time = 0

	resources = {
		category = decisions
	}

	potential = {
		exists = from
		from = {
			is_ai = no
			has_country_flag = yykex
        }
		is_star = yes
        solar_system = { is_capital_system = yes }
	}

    effect = {
        create_fleet = {
            name = random
            settings = {
                can_upgrade = yes
                can_change_composition = yes
                can_change_leader = yes
                spawn_debris = yes
                uses_naval_capacity = yes
                can_disband = yes
                is_boss = no
                is_ultra_boss = no
            }
            effect = {
                set_owner = root.from
                while = {
                    count = 5
                    create_ship = {
                        name = random
                        design = "NAME_Savant"
                        upgradable = yes
                        graphical_culture = "fallen_empire_01"
                        prefix = no
                    }
                }
                while = {
                    count = 30
                    create_ship = {
                        name = random
                        design = "NAME_Scholar"
                        upgradable = yes
                        graphical_culture = "fallen_empire_01"
                        prefix = no
                    }
                }
                while = {
                    count = 25
                    create_ship = {
                        name = random
                        design = "NAME_Sage"
                        upgradable = yes
                        graphical_culture = "fallen_empire_01"
                        prefix = no
                    }
                }
                set_formation_scale = 1
                set_location = root
            }
        }
        create_fleet = {
            name = random
            settings = {
                can_upgrade = yes
                can_change_composition = yes
                can_change_leader = yes
                spawn_debris = yes
                uses_naval_capacity = yes
                can_disband = yes
                is_boss = no
                is_ultra_boss = no
            }
            effect = {
                set_owner = root.from
                while = {
                    count = 10
                    create_ship = {
                        name = random
                        design = "NAME_Scholar"
                        upgradable = yes
                        graphical_culture = "fallen_empire_01"
                        prefix = no
                    }
                }
                set_formation_scale = 1
                set_location = root
            }
        }
    }
}
