yyk_leader_trait_change = {
	sound = event_default
	icon = default
	enactment_time = 0

	resources = {
		category = decisions
	}

	potential = {
		exists = owner
		owner = {
			is_ai = no
			has_country_flag = yykex
		}
        is_capital = yes
	}

	effect = {
		owner = {
			every_pool_leader = {
				limit = { NOT = { has_leader_flag = yykex_leader_trait_changed } }
				if = {
					limit = {
						NOT = { species = { has_trait = trait_machine_unit } }
					}
					if = {
						limit = {
							OR = {
								has_trait = leader_trait_scholarium_investigator
								has_trait = leader_trait_scholarium_explorer
								has_trait = leader_trait_scholarium_observer
							}
						}
						if = {
							limit = { years_passed < 10 }
							remove_all_traits = yes
							add_trait = leader_trait_scholarium_explorer
						}
						else = {
							remove_all_traits = yes
							add_trait = leader_trait_scholarium_investigator
						}
					}
					else_if = {
						limit = {
							OR = {
								has_trait = leader_trait_bulwark_watcher
								has_trait = leader_trait_bulwark_retaliator
								has_trait = leader_trait_bulwark_sentinel
							}
						}
						remove_all_traits = yes
						add_trait = leader_trait_bulwark_retaliator
					}
					else_if = {
						limit = {
							OR = {
								has_trait = leader_trait_prospectorium_extractor
								has_trait = leader_trait_prospectorium_optimizer
								has_trait = leader_trait_prospectorium_refiner
							}
						}
						remove_all_traits = yes
						add_trait = leader_trait_prospectorium_optimizer
					}
					if = {
						limit = { species = { has_trait = trait_erudite } }
						add_trait = leader_trait_erudite
					}
					if = {
						limit = {species = { has_trait = trait_cybernetic } }
						add_trait = leader_trait_cyborg
					}
					if = {
						limit = {species = { has_trait = trait_psionic } }
						add_trait = leader_trait_psionic
					}
				}
				else = {
					remove_all_traits = yes
					add_trait = leader_trait_synthetic
					add_trait = leader_trait_psionic
				}
				set_leader_flag = yykex_leader_trait_changed
				set_skill = 10
				add_trait = leader_trait_eager
				add_trait = leader_trait_resilient
				add_trait = leader_trait_emotional_support_pet
				change_background_ethic = ethic_materialist
				set_variable = {
					which = "leader_pending_negative_traits_unmodified"
					value = 0
				}
				if = {
					limit = { leader_class = scientist }
					add_trait = leader_trait_roamer_2
					add_trait = leader_trait_archaeologist_2
					add_trait = leader_trait_meticulous_2
					add_trait = leader_trait_sapient_AI_assistant
					add_trait = leader_trait_planar_theorist_2
					add_trait = leader_trait_void_tracker_2
					add_trait = leader_trait_scientist_mind_wipe_02
					if = {
						limit = { years_passed < 10}
						add_trait = subclass_scientist_explorer
						add_trait = leader_trait_prospector_3
						add_trait = leader_trait_observant_2 
						add_trait = leader_trait_xeno_cataloger
					}
					else = {
						add_trait = subclass_scientist_governor
						add_trait = leader_trait_environmental_engineer
						add_trait = leader_trait_architectural_interest
						add_trait = leader_trait_collaborator_3
						add_trait = leader_trait_physics_focus_2
						add_trait = leader_trait_robotist
					}
				}
				if = {
					limit = { leader_class = commander }
					add_trait = subclass_commander_admiral
					add_trait = leader_trait_cautious
					add_trait = leader_trait_unyielding
					add_trait = leader_trait_engineer
					add_trait = leader_trait_dragonslayer
					add_trait = leader_trait_restrained
					add_trait = leader_trait_admiral_mind_wipe_02
					add_trait = leader_trait_void_hunter_2
					add_trait = leader_trait_commanding_presence_2
					add_trait = leader_trait_artillery_specialization_2
					add_trait = leader_trait_artillerist 
					add_trait = leader_trait_master_gunner
				}
				if = {
					limit = { leader_class = official }
					add_trait = subclass_official_governor
					add_trait = leader_trait_righteous_2
					add_trait = leader_trait_environmental_engineer
					add_trait = leader_trait_enlister
					add_trait = leader_trait_architectural_interest
					add_trait = leader_trait_trade_focus_2
					add_trait = leader_trait_pop_focused_3
					add_trait = leader_trait_assembler_2
					add_trait = leader_trait_robotist
				}
			}
		}
	}

	ai_weight = {
		weight = 0
	}
}

yyk_leader_negative_traits_prevent = {
	sound = event_default
	icon = default
	enactment_time = 0

	resources = {
		category = decisions
	}

	potential = {
		exists = owner
		owner = {
			is_ai = no
			OR = {
				has_country_flag = yykex
				has_country_flag = yykex_friend
			}
		}
        is_capital = yes
	}

	effect = {
		owner = {
			every_owned_leader = {
				set_variable = {
					which = "leader_pending_negative_traits_unmodified"
					value = 0
				}
			}
		}
	}

}
